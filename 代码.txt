        AOS.init({ offset: 120, duration: 800, easing: 'ease-out-cubic', once: true });
        
        const langMap = {
            'en': { class: 'lang-en', label: 'English' },
            'ar': { class: 'lang-ar', label: 'العربية' },
            'zh': { class: 'lang-zh', label: '中文' }
        };

        function toggleDropdown(event) {
            event.stopPropagation();
            const menu = document.getElementById('langMenu');
            menu.classList.toggle('show');
            document.querySelector('.lang-btn').classList.toggle('active');
        }

        function setLang(langCode) {
            document.body.classList.remove('lang-en', 'lang-ar', 'lang-zh');
            const target = langMap[langCode];
            document.body.classList.add(target.class);
            document.getElementById('current-lang').innerText = target.label;
            
            const menu = document.getElementById('langMenu');
            menu.classList.remove('show');
            document.querySelector('.lang-btn').classList.remove('active');
            
            setTimeout(function(){ AOS.refresh(); }, 100);
        }

        window.onclick = function(event) {
            if (!event.target.closest('.lang-dropdown')) {
                const menu = document.getElementById('langMenu');
                if (menu.classList.contains('show')) {
                    menu.classList.remove('show');
                    document.querySelector('.lang-btn').classList.remove('active');
                }
            }
        }

        const counters = document.querySelectorAll('.counter');
        counters.forEach(counter => {
            const updateCount = () => {
                const target = +counter.getAttribute('data-target') || +counter.innerText;
                const count = +counter.innerText;
                const inc = target / 100;
                if (count < target) {
                    counter.innerText = Math.ceil(count + inc);
                    setTimeout(updateCount, 20);
                } else {
                    counter.innerText = target;
                }
            };
            setTimeout(updateCount, 1000);
        });

        const brands = document.querySelectorAll('#brand-rotator img');
        if(brands.length > 0) {
            let currentBrand = 0;
            setInterval(() => {
                brands[currentBrand].classList.remove('active');
                currentBrand = (currentBrand + 1) % brands.length;
                brands[currentBrand].classList.add('active');
            }, 3000); 
        }
    </script>
